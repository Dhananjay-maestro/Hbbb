  <script>
    /* -------- confetti canvas -------- */
    const canvas = document.getElementById('confetti');
    const ctx = canvas.getContext('2d');
    let W, H, pieces = [], animId;

    function fitCanvas(){
      canvas.width = canvas.offsetWidth;
      canvas.height = canvas.offsetHeight;
      W = canvas.width; H = canvas.height;
    }
    window.addEventListener('resize', fitCanvas);
    fitCanvas();

    function rand(a,b){ return Math.random()*(b-a)+a }

    function spawn(n = 140){
      for(let i=0;i<n;i++){
        pieces.push({
          x: rand(0, W),
          y: rand(-H, 0),
          w: rand(6, 14),
          h: rand(8, 18),
          r: rand(0, 360),
          velX: rand(-1.6, 1.6),
          velY: rand(2, 6),
          rotSpeed: rand(-3,3),
          color: ['#ff2d7a','#ffcf4d','#8de0a6','#00bcd4','#ff8bcf'][Math.floor(rand(0,5))],
          tilt: rand(-15,15)
        });
      }
      if(!animId) animate();
    }

    function drawPieces(){
      ctx.clearRect(0,0,W,H);
      for(let i=0;i<pieces.length;i++){
        const p = pieces[i];
        p.x += p.velX; p.y += p.velY; p.r += p.rotSpeed;
        // draw rounded rect-ish confetti
        ctx.save();
        ctx.translate(p.x, p.y);
        ctx.rotate(p.r * Math.PI / 180);
        ctx.fillStyle = p.color;
        ctx.fillRect(-p.w/2, -p.h/2, p.w, p.h);
        ctx.restore();
      }
      // remove off-screen
      pieces = pieces.filter(p => p.y < H + 40);
      if(pieces.length === 0){ cancelAnimationFrame(animId); animId = null; ctx.clearRect(0,0,W,H) }
    }
    function animate(){ animId = requestAnimationFrame(animate); drawPieces(); }

    /* -------- sparkles DOM -------- */
    const sparkles = document.getElementById('sparkles');
    function makeSparkles(n=16){
      for(let i=0;i<n;i++){
        const d = document.createElement('div');
        d.className = 'spark';
        const left = 8 + Math.random()*84;
        const top = 8 + Math.random()*84;
        d.style.left = left + '%';
        d.style.top = top + '%';
        d.style.opacity = 0;
        sparkles.appendChild(d);
        const dur = 700 + Math.random()*900;
        d.animate([
          {opacity:0, transform:'scale(.6)'},
          {opacity:1, transform:'scale(1.1)'},
          {opacity:0, transform:'scale(.2)'}
        ], { duration: dur, easing:'cubic-bezier(.2,.9,.2,1)'});
        setTimeout(()=> d.remove(), dur);
      }
    }

    /* attach actions */
    document.getElementById('partyBtn').addEventListener('click', ()=>{ spawn(180); makeSparkles(26) });

    document.getElementById('photoBtn').addEventListener('click', ()=>{
      // simple placeholder ‚Äî user can replace with file input or direct image insertion
      const url = prompt('Paste image URL of Princee (jpg/png) ‚Äî ya phir "cancel" kar dein:');
      if(url){ // insert a circular photo on left illustration
        const img = document.createElement('img');
        img.src = url;
        img.alt = 'Princee photo';
        img.style.position = 'absolute';
        img.style.left = '50%';
        img.style.bottom = '12%';
        img.style.transform = 'translateX(-50%)';
        img.style.width = '140px';
        img.style.height = '140px';
        img.style.objectFit = 'cover';
        img.style.borderRadius = '18px';
        img.style.boxShadow = '0 12px 34px rgba(0,0,0,0.18)';
        const visual = document.querySelector('.stage-illustration');
        // remove old user photo if exists
        document.querySelectorAll('.stage-illustration .user-photo').forEach(e=>e.remove());
        img.className = 'user-photo';
        visual.appendChild(img);
      }
    });

    /* music toggle */
    const audio = document.getElementById('bgAudio');
    const musicBtn = document.getElementById('musicToggle');
    let musicOn = false;
    musicBtn.addEventListener('click', ()=>{
      if(!musicOn){
        audio.play().catch(()=>{ alert('Browser blocked autoplay. Press Play on your device to allow audio.'); });
        musicBtn.textContent = 'Pause Music ‚úï';
        musicOn = true;
      } else {
        audio.pause(); audio.currentTime = 0;
        musicBtn.textContent = 'Play Music ‚ô™';
        musicOn = false;
      }
    });

    // preload images to avoid pop-in
    ['https://i.imgur.com/2KQpVQv.png','https://i.imgur.com/6Y0Qb6u.png','https://i.imgur.com/5Y3Xb9G.png','https://i.imgur.com/0K7Y3vD.png']
      .forEach(u=>{ const i=new Image(); i.src=u });
